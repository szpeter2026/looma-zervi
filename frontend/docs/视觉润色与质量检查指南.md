## 2. 视觉润色实施指南

### 2.1 微交互动画润色

```css
/* 现有基础 */
--px-anim-fast: 150ms ease;
--px-anim-normal: 250ms ease;

/* 建议添加的动画Token */
--px-anim-bounce: 400ms cubic-bezier(0.68, -0.55, 0.265, 1.55);
--px-anim-spring:卷二 500ms cubic-bezier(0.34, 1.56, 0.64, 1);
--px-anim-elastic: 600ms cubic-bezier(0.68, -0.55, 0.265, 1.55);
```

**应用示例**：
```tsx
// 按钮点击效果
<PlanetXButton
  onClick={handleClick}
  style={{
    transition: "transform var(--px-anim-spring), background var(--px-anim-fast)",
  }}
  onMouseDown={(e) => e.currentTarget.style.transform = "scale(0.95)"}
  onMouseUp={(e) => e.currentTarget.style.transform = "scale(1)"}
>
  Click Me
</PlanetXButton>

// 卡片悬停效果
<PlanetXCard
  style={{
    transition: "transform var(--px-anim-normal), box-shadow var(--px-anim-normal)",
  }}
  onMouseEnter={(e) => {
    e.currentTarget.style.transform = "translateY(-4px)";
    e.currentTarget.style.boxShadow = "var(--px-shadow-card-hover)";
  }}
  onMouseLeave={(e) => {
    e.currentTarget.style.transform = "translateY(0)";
    e.currentTarget.style.boxShadow = "var(--px-shadow-card)";
  }}
>
  Content
</PlanetXCard>
```

### 3.2 设计Token合规检查

```javascript
// scripts/check-tokens.js
const fs = require('fs');
const path = require('path');

function checkTokenUsage(filePath) {
  const content = fs.readFileSync(filePath, 'utf8');
  const hardcodedColors = content.match(/#[0-9a-fA-F]{3,6}/g);
  const hardcodedSpacing = content.match(/[0-9]+px/g);
  
  return {
    file: filePath,
    hardcodedColors: hardcodedColors || [],
    hardcodedSpacing: hardcodedSpacing || [],
    usesTokens: content.includes('var(--'),
  };
}

// 运行检查
const results = checkTokenUsage('src/components/Example.tsx');
console.log('设计Token合规检查结果:', results);
```
